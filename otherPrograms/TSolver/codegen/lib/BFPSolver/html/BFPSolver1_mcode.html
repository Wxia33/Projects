1function Out = BFPSolver(xmin,xmax)  
 2%#codegen
 3coder.inline('never');
 4
 5% Bisection False Position Method Algorithm
 6% Find the roots of F(x) = 0 given a search interval [xmin xmax];
 7tole = 1e-6;     % Tolerance level
 8
 9% Make sure the function has at least one zero
10ffm=fun(xmin);
11ffM=fun(xmax);
12if (ffm > 0 && ffM > 0) || (ffm < 0 && ffM < 0)
13    error('The function might not have zeros');
14end
15
16% Bisection algortihm
17iter=1;
18% Loop
19if ffm > 0 
20    % Decrea function
21    xmid = 0.5*(xmin+xmax);
22    fmid = fun(xmid);
23    while ((ffm-ffM) > tole) && abs(fmid)>tole
24        xmid = xmin - ((xmax-xmin)/(ffM - ffm)) *ffm;
25        fmid = fun(xmid);
26        if fmid > 0
27            xmin = xmid;
28            ffm=fun(xmin);
29        else
30            xmax = xmid;
31             ffM=fun(xmax);
32        end
33        iter=iter+1;
34    end
35else
36    % Increas function
37    xmid = 0.5*(xmin+xmax);
38    fmid = fun(xmid);
39    while (ffM-ffm) > tole  && abs(fmid)>tole
40        xmid = xmin - ((xmax-xmin)/(ffM - ffm)) *ffm;
41        fmid = fun(xmid);
42        if fmid > 0
43            xmax = xmid;
44            ffM=fun(xmax);
45        else
46            xmin = xmid;
47            ffm=fun(xmin);
48        end
49        iter=iter+1;
50    end
51end
52
53Out = xmid;
54end
